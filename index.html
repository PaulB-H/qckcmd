Server Block Example

Sections managed by certbot are inserted upon SSL generation

The first section listens on port 443 for HTTPS and serves index/root dir


The second section listens on port 80 for HTTP requests, and redirects them to HTTPS 

server {
  # Used to select server block for processing
  server_name mydomain.ca www.mydomain.ca;

  # Configures publicly served root dir
  root /var/www/mydomain;
	
  # Configures index file to be served
  index index.html index.htm;
	
  # Below managed by Certbot
  listen 443 ssl;
  ssl_certificate /etc/letsencrypt/live/www.mydomain.com/fullchain.pem;
  ssl_certificate_key /etc/letsencrypt/live/www.mydomain.com/privkey.pem;
  include /etc/letsencrypt/options-ssl-nginx.conf;
  ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;
} 
	
server {
  # Redirect if not https
  if ($host = www.mydomain.ca) {
    return 301 https://$host$request_uri;
  } # Managed by Certbot
  if ($host = mydomain.ca) {
    return 301 https://$host$request_uri;
  } # Managed by Certbot
	
  listen 80;
  server_name mydomain.ca www.mydomain.ca;
	
  return 404; # managed by Certbot
	
}

Server Block with proxy pass

With this configuration, requests for 'www.mydomain.com/server1' will serve whatever is running on localhost:5000, whereas 'www.mydomain.com/server2' will serve localhost:6000

This is a good example for how you can have multiple Node.js servers running, on a single VPS, even under a single domain. 

server {
  server_name mydomain.com www.mydomain.com;

  # Below managed by Certbot
  listen 443 ssl;
  ssl_certificate /etc/letsencrypt/live/www.mydomain.com/fullchain.pem;
  ssl_certificate_key /etc/letsencrypt/live/www.mydomain.com/privkey.pem;
  include /etc/letsencrypt/options-ssl-nginx.conf;
  ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

  location /server1  {
    proxy_pass    http://localhost:5000;
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection 'upgrade';
    proxy_set_header Host $host;
    proxy_cache_bypass $http_upgrade;
  }

  location /server2  {
    proxy_pass    http://localhost:6000;
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection 'upgrade';
    proxy_set_header Host $host;
    proxy_cache_bypass $http_upgrade;
  }

}
server {
  if ($host = www.mydomain.com) {
    return 301 https://$host$request_uri;
  } # managed by Certbot

  if ($host = mydomain.com) {
    return 301 https://$host$request_uri;
  } # managed by Certbot

  listen 80;
  server_name mydomain.com www.mydomain.com;
  return 404; # managed by Certbot

}

Sorting the above...

...This was easier than doing it manually, and allows me to keep appending packages without thinking about order. 

const sortPackageList = () => {
  const packages = document.getElementById("package-list");
  let len = packages.children.length;
  let IDarr = [];
  let packageArr = [];

  for (let i = 0; i < len; i++) {
    IDarr.push(packages.children[i].id);
    packageArr.push(packages.children[i]);
  }
  IDarr.sort();

  let sortedElems = [];
  IDarr.forEach((id) => {
    packageArr.forEach((elem) => {
      if (elem.id === id) {
        sortedElems.push(elem);
      }
    });
  });

  packages.innerHTML = "";
  sortedElems.forEach((item) =>
    packages.insertAdjacentElement("beforeend", item)
  );
};
